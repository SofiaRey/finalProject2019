Base de datos 

create database Proyecto;
use Proyecto;



Tablas

create table usuario(
    CI integer, 
    id integer UNIQUE, 
    nombre varchar(50), 
    apellido varchar(50), 
    mail varchar(50) UNIQUE, 
    password varchar(40), 
    primary key(CI)
);
create table bibliotecario(
    CI integer PRIMARY KEY,
    FOREIGN KEY(CI) references usuario(CI)
);  
create table estudiante(
    CI integer PRIMARY KEY, 
    orientacion varchar(3), 
    foreign key(CI) references usuario(CI)
);  
create table Profesor(
    CI integer PRIMARY KEY, 
    orientacion varchar(7), 
    foreign key(CI) references usuario(CI)
); 
create table prestamo(
    id integer , 
    devuelto bit, 
    fechaDev date, 
    fechaSol date, 
    ciUsuario integer, 
    idLibro varchar(50), 
    primary key(ciUsuario,idLibro), 
    foreign key(ciUsuario) references usuario(CI), 
    foreign key(idLibro) references libro(CodLibro)
);
create table libro (
    CodLibro varchar(50) UNIQUE, 
    CodISBN varchar(50) UNIQUE, 
    nroEdicion integer,
    genero varchar(20), 
    autor varchar(50), 
    añoPub smallint, 
    descripcion text, 
    editorial varchar(50), 
    URLcover varchar(50), 
    cant integer, 
    disponible bit,
    primary key(CodLibro,CodISBN)
);
