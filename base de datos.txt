Base de datos 

create database Proyecto;
use Proyecto;



Tablas

create table usuario(
    CI integer  NOT NULL, 
    id integer UNIQUE  NOT NULL, 
    nombre varchar(50)  NOT NULL, 
    apellido varchar(50)  NOT NULL, 
    mail varchar(50) UNIQUE  NOT NULL, 
    password varchar(40)  NOT NULL, 
    primary key(CI)
);
create table bibliotecario(
    CI integer PRIMARY KEY,
    FOREIGN KEY(CI) references usuario(CI)
);  
create table estudiante(
    CI integer PRIMARY KEY, 
    orientacion varchar(3)  NOT NULL, 
    foreign key(CI) references usuario(CI)
);  
create table Profesor(
    CI integer PRIMARY KEY, 
    orientacion varchar(7)  NOT NULL, 
    foreign key(CI) references usuario(CI)
); 
create table prestamo(
    id integer  NOT NULL, 
    devuelto bit  NOT NULL, 
    fechaDev date  NOT NULL, 
    fechaSol date  NOT NULL, 
    ciUsuario integer  NOT NULL, 
    idLibro varchar(50)  NOT NULL, 
    primary key(ciUsuario,idLibro), 
    foreign key(ciUsuario) references usuario(CI), 
    foreign key(idLibro) references libro(CodLibro)
);
create table libro (
    CodLibro varchar(50) UNIQUE NOT NULL, 
    CodISBN varchar(50) UNIQUE  NOT NULL, 
    nroEdicion integer  NOT NULL,
    genero varchar(20)  NOT NULL, 
    autor varchar(50)  NOT NULL, 
    añoPub smallint  NOT NULL, 
    descripcion text  NOT NULL, 
    editorial varchar(50)  NOT NULL, 
    URLcover varchar(50)  NOT NULL, 
    cant integer  NOT NULL,
    hayEjemplaresDisponibles bit NOT NULL,
    cantEjemplaresDisp integer  NOT NULL, 
    disponible bit  NOT NULL,
    primary key(CodLibro)
);
